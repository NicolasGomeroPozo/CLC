<body>


<div id="temporizador">05:00</div>

<button id="startBtn">Start</button>

<button id="retryBtn" style="display:none;">Retry</button>


<script>

    var temporizador;

    var minutes = 5;

    var seconds = 0;


    document.getElementById('startBtn').addEventListener('click', function() {

        if (!temporizador) {

            // Inicia el temporizador

            this.style.display = 'none'; // Oculta el botón de inicio

            document.getElementById('retryBtn').style.display = 'inline'; // Muestra el botón de reinicio

            temporizador = setInterval(actualizarTemporizador, 1000);

        }

    });


    document.getElementById('retryBtn').addEventListener('click', function() {

        clearInterval(temporizador);

        temporizador = null;

        minutes = 5;

        seconds = 0;

        document.getElementById('temporizador').textContent = "05:00";

        document.getElementById('startBtn').style.display = 'inline'; // Muestra el botón de inicio

        this.style.display = 'none'; // Oculta el botón de reinicio

    });


    function actualizarTemporizador() {

        seconds--;

        if (seconds < 0) {

            seconds = 59;

            minutes--;

        }


        document.getElementById('temporizador').textContent = 

            (minutes > 9 ? minutes : "0" + minutes) + ":" + 

            (seconds > 9 ? seconds : "0" + seconds);


        if (minutes < 0) {

            clearInterval(temporizador);

            temporizador = null;

            document.getElementById('temporizador').textContent = "00:00";

            alert("¡Tiempo terminado!");

            document.getElementById('retryBtn').click(); // Reinicia el temporizador automáticamente

        }

    }

</script>


</body>
